41/// Represents errors that can occur in the GolemBase ETH client.
42/// Used for wrapping transaction, receipt, and log decoding errors.
43#[derive(Debug, Display, Error)]
44pub enum Error {
45    /// Failed to send transaction: {0}
46    TransactionSendError(String),
47    /// Failed to get transaction receipt: {0}
48    TransactionReceiptError(String),
49    /// Failed to decode expiration block: {0}
50    ExpirationBlockDecodeError(String),
51    /// Unexpected log data format
52    UnexpectedLogDataError,
53}
54
55/// The Ethereum address of the GolemBase storage contract.
56/// All entity-related transactions are sent to this address.
57pub const STORAGE_ADDRESS: Address = address!("0x0000000000000000000000000000000060138453");

59impl GolemBaseClient {
60    pub async fn send_transaction(
61        &self,
62        creates: Vec<Create>,
63        updates: Vec<Update>,
64        deletes: Vec<Hash>,
65        extensions: Vec<Extend>,
66    ) -> Result<TransactionResult, Error> {
67        let receipt = self
68            .create_raw_transaction(GolemBaseTransaction {
69                creates,
70                updates,
71                deletes,
72                extensions,
73            })
74            .await?;
75
76        let mut txres = TransactionResult::default();
77        receipt.logs().iter().cloned().try_for_each(|log| {
78            let log: alloy::primitives::Log = log.into();
79            let parsed = GolemBaseABI::GolemBaseABIEvents::decode_log(&log).map_err(|e| {
80                Error::TransactionReceiptError(format!("Error decoding event log: {}", e))
81            })?;
82            match parsed.data {
83                GolemBaseABI::GolemBaseABIEvents::GolemBaseStorageEntityCreated(data) => {
84                    txres.creates.push(EntityResult {
85                        entity_key: data.entityKey.into(),
86                        expiration_block: data.expirationBlock.try_into().unwrap_or_default(),
87                    });
88                    Ok(())
89                }
90                GolemBaseABI::GolemBaseABIEvents::GolemBaseStorageEntityUpdated(data) => {
91                    txres.updates.push(EntityResult {
92                        entity_key: data.entityKey.into(),
93                        expiration_block: data.expirationBlock.try_into().unwrap_or_default(),
94                    });
95                    Ok(())
96                }
97                GolemBaseABI::GolemBaseABIEvents::GolemBaseStorageEntityDeleted(data) => {
98                    txres.deletes.push(DeleteResult {
99                        entity_key: data.entityKey.into(),
100                    });
101                    Ok(())
102                }
103                GolemBaseABI::GolemBaseABIEvents::GolemBaseStorageEntityBTLExtended(data) => {
104                    txres.extensions.push(ExtendResult {
105                        entity_key: data.entityKey.into(),
106                        old_expiration_block: data
107                            .oldExpirationBlock
108                            .try_into()
109                            .unwrap_or_default(),
110                        new_expiration_block: data
111                            .newExpirationBlock
112                            .try_into()
113                            .unwrap_or_default(),
114                    });
115                    Ok(())
116                }
117            }
118        })?;
119
120        Ok(txres)
121    }

123    /// Creates one or more new entities in GolemBase and returns their results.
124    /// Sends a transaction to the storage contract and parses the resulting logs.
125    pub async fn create_entities(&self, creates: Vec<Create>) -> Result<Vec<EntityResult>, Error> {
126        let result = self.send_transaction(creates, vec![], vec![], vec![]).await;
127        result.and_then(|res| match res {
128            TransactionResult {
129                creates,
130                updates,
131                deletes,
132                extensions,
133            } if updates.is_empty() && deletes.is_empty() && extensions.is_empty() => Ok(creates),
134            _ => Err(Error::TransactionReceiptError(
135                "Unexpected content in tx logs, this should never happen!".to_string(),
136            )),
137        })
138    }

196    /// Creates and sends a raw transaction to the GolemBase storage contract.
197    /// Encodes the transaction payload and sends it to the contract address.
198    ///
199    /// NOTE: Nonce management is tricky!
200    /// - This implementation always tries to fetch the latest on-chain nonce before sending a transaction,
201    ///   and only falls back to the locally cached base_nonce if the sync fails.
202    /// - Only the number of in-flight transactions is tracked locally.
203    /// - For robust production use, consider also handling stuck transactions (e.g., gas bumping/EIP-1559).
204    pub async fn create_raw_transaction(
205        &self,
206        payload: GolemBaseTransaction,
207    ) -> Result<TransactionReceipt, Error> {
208        log::debug!("payload: {:?}", payload);
209        let mut buffer = Vec::new();
210        payload.encode(&mut buffer);
211        log::debug!("buffer: {:?}", buffer);
212        let nonce = {
213            // This is sadly needed because `self.provider.get_transaction_count(self.wallet.address()).pending()`
214            // doesn't give the right number...
215            //
216            //      Error: server returned an error response: error code -32000: replacement transaction underpriced
217            let mut nm = self.nonce_manager.lock().await;
218            let wallet_address = self.wallet.address();
219            match self.provider.get_transaction_count(wallet_address).await {
220                Ok(on_chain_nonce) => {
221                    nm.base_nonce = on_chain_nonce;
222                }
223                Err(e) => {
224                    log::warn!("Failed to fetch on-chain nonce: {}", e);
225                }
226            }
227            nm.next_nonce().await
228        };
229        let tx_base = TransactionRequest {
230            to: Some(TxKind::Call(STORAGE_ADDRESS)),
231            input: buffer.into(),
232            chain_id: Some(
233                self.provider
234                    .get_chain_id()
235                    .await
236                    .map_err(|e| Error::TransactionSendError(e.to_string()))?,
237            ),
238            nonce: Some(nonce),
239            ..Default::default()
240        };
241        log::debug!("transaction: {:?}", tx_base);
242        let estimated_gas = self
243            .provider
244            .estimate_gas(tx_base.clone())
245            .await
246            .map_err(|e| Error::TransactionSendError(format!("Failed to estimate gas: {}", e)))?;
247        let tx = tx_base.with_gas_limit(estimated_gas).with_gas_price(
248            self.provider
249                .get_gas_price()
250                .await
251                .map_err(|e| Error::TransactionSendError(e.to_string()))?,
252        );
253        let pending_tx = self
254            .provider
255            .send_transaction(tx)
256            .await
257            .map_err(|e| Error::TransactionSendError(e.to_string()))?;
258        log::debug!("pending transaction: {:?}", pending_tx);
259        let receipt = pending_tx
260            .get_receipt()
261            .await
262            .map_err(|e| Error::TransactionReceiptError(e.to_string()))?;
263        log::debug!("receipt: {:?}", receipt);
264        {
265            let mut nm = self.nonce_manager.lock().await;
266            nm.complete().await;
267        }
268        Ok(receipt)
269    }
