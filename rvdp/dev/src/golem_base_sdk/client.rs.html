16/// Tracks and assigns sequential Ethereum nonces for concurrent transactions.
17pub struct NonceManager {
18    /// Last known on-chain nonce.
19    pub base_nonce: u64,
20    /// Number of in-flight (pending) transactions.
21    pub in_flight: u64,
22}
23
24impl NonceManager {
25    /// Returns the next available nonce and increments the in-flight counter.
26    pub async fn next_nonce(&mut self) -> u64 {
27        let nonce = self.base_nonce + self.in_flight;
28        self.in_flight += 1;
29        nonce
30    }
31
32    /// Marks a transaction as completed by decrementing the in-flight counter.
33    pub async fn complete(&mut self) {
34        if self.in_flight > 0 {
35            self.in_flight -= 1;
36        }
37    }
38}

62/// A client for interacting with the GolemBase system.
63/// Provides methods for account management, entity operations, balance queries, and event subscriptions.
64#[derive(Clone)]
65pub struct GolemBaseRoClient {
66    /// The underlying provider for making RPC calls.
67    pub(crate) provider: DynProvider,
68}
69
70#[bon]
71impl GolemBaseRoClient {
72    /// Creates a new builder for `GolemBaseClient` with the given wallet and RPC URL.
73    /// Initializes the provider and sets up default configuration.
74    #[builder]
75    pub fn builder(rpc_url: Url, provider: Option<DynProvider>) -> Self {
76        let provider = provider.unwrap_or_else(|| {
77            ProviderBuilder::new()
78                .connect_http(rpc_url.clone())
79                .erased()
80        });
81
82        Self { provider }
83    }
84}

86/// A client for interacting with the GolemBase system.
87/// Provides methods for account management, entity operations, balance queries, and event subscriptions.
88#[derive(Clone)]
89pub struct GolemBaseClient {
90    /// The underlying GolemBaseRoClient
91    pub(crate) ro_client: GolemBaseRoClient,
92    /// The Ethereum address of the client owner.
93    pub(crate) wallet: PrivateKeySigner,
94    /// Nonce manager for tracking transaction nonces.
95    pub(crate) nonce_manager: Arc<Mutex<NonceManager>>,
96}
97
98impl Deref for GolemBaseClient {
99    type Target = GolemBaseRoClient;
100
101    fn deref(&self) -> &Self::Target {
102        &self.ro_client
103    }
104}
105
106#[bon]
107impl GolemBaseClient {
108    /// Creates a new builder for `GolemBaseClient` with the given wallet and RPC URL.
109    /// Initializes the provider and sets up default configuration.
110    #[builder]
111    pub fn builder(wallet: PrivateKeySigner, rpc_url: Url) -> Self {
112        let provider = ProviderBuilder::new()
113            .wallet(wallet.clone())
114            .connect_http(rpc_url.clone())
115            .erased();
116
117        let ro_client = GolemBaseRoClient::builder()
118            .rpc_url(rpc_url)
119            .provider(provider)
120            .build();
121
122        Self {
123            ro_client,
124            wallet,
125            nonce_manager: Arc::new(Mutex::new(NonceManager {
126                base_nonce: 0,
127                in_flight: 0,
128            })),
129        }
130    }
131
132    /// Gets the underlying Reqwest client used for HTTP requests.
133    pub fn get_reqwest_client(&self) -> ClientRef<'_> {
134        self.provider.client()
135    }
136
137    /// Gets the Ethereum address of the client owner.
138    pub fn get_owner_address(&self) -> Address {
139        self.wallet.address()
140    }
141
142    /// Gets the chain ID from the provider.
143    /// Returns the chain ID as a `u64`.
144    pub async fn get_chain_id(&self) -> anyhow::Result<u64> {
145        self.provider
146            .get_chain_id()
147            .await
148            .map_err(|e| anyhow::anyhow!("Failed to get chain ID: {}", e))
149    }
150
151    /// Gets an account's ETH balance as a `BigDecimal`.
152    pub async fn get_balance(&self, account: Address) -> anyhow::Result<BigDecimal> {
153        let balance = self.provider.get_balance(account).await?;
154        Ok(wei_to_eth(balance))
155    }
156
157    /// Gets the current block number from the chain.
158    /// Returns the latest block number as a `u64`.
159    pub async fn get_current_block_number(&self) -> anyhow::Result<u64> {
160        let latest_block = self
161            .provider
162            .get_block_by_number(BlockNumberOrTag::Latest)
163            .await?
164            .ok_or_else(|| anyhow::anyhow!("Failed to get latest block"))?;
165        Ok(latest_block.header.number)
166    }
167}
