16/// Represents a GolemBase event parsed from the blockchain log.
17/// Used to distinguish between entity creation, update, and removal events.
18#[derive(Debug)]
19pub enum Event {
20    /// Entity was created.
21    /// Contains the entity ID, block number, and transaction hash.
22    EntityCreated {
23        /// The ID of the created entity
24        entity_id: Hash,
25        /// The expiration block of the entity
26        expiration_block: u64,
27        /// The block number where the event occurred
28        block_number: u64,
29        /// The transaction hash that triggered the event
30        transaction_hash: Hash,
31    },
32    /// Entity was updated.
33    /// Contains the entity ID, block number, and transaction hash.
34    EntityUpdated {
35        /// The ID of the updated entity
36        entity_id: Hash,
37        /// The expiration block of the entity
38        expiration_block: u64,
39        /// The block number where the event occurred
40        block_number: u64,
41        /// The transaction hash that triggered the event
42        transaction_hash: Hash,
43    },
44    /// Entity was removed.
45    /// Contains the entity ID, block number, and transaction hash.
46    EntityRemoved {
47        /// The ID of the removed entity
48        entity_id: Hash,
49        /// The block number where the event occurred
50        block_number: u64,
51        /// The transaction hash that triggered the event
52        transaction_hash: Hash,
53    },
54    /// Entity was extended.
55    /// Contains the entity ID, block number, and transaction hash.
56    EntityExtended {
57        /// The ID of the removed entity
58        entity_id: Hash,
59        /// The old expiration block
60        old_expiration_block: u64,
61        /// The new expiration block
62        new_expiration_block: u64,
63        /// The block number where the event occurred
64        block_number: u64,
65        /// The transaction hash that triggered the event
66        transaction_hash: Hash,
67    },
68}

70impl TryFrom<Log> for Event {
71    type Error = anyhow::Error;
72
73    /// Attempts to parse a blockchain log into a `Event`.
74    /// Returns an error if required fields are missing or the event type is unknown.
75    fn try_from(log: Log) -> Result<Self> {
76        let block_number = log
77            .block_number
78            .ok_or_else(|| anyhow::anyhow!("Missing block number"))?;
79        let transaction_hash = log
80            .transaction_hash
81            .ok_or_else(|| anyhow::anyhow!("Missing transaction hash"))?;
82        let parsed = GolemBaseABI::GolemBaseABIEvents::decode_log(&log.into())?;
83        match parsed.data {
84            GolemBaseABI::GolemBaseABIEvents::GolemBaseStorageEntityCreated(data) => {
85                Ok(Event::EntityCreated {
86                    entity_id: data.entityKey.into(),
87                    expiration_block: data.expirationBlock.try_into().unwrap_or_default(),
88                    block_number,
89                    transaction_hash,
90                })
91            }
92            GolemBaseABI::GolemBaseABIEvents::GolemBaseStorageEntityUpdated(data) => {
93                Ok(Event::EntityUpdated {
94                    entity_id: data.entityKey.into(),
95                    expiration_block: data.expirationBlock.try_into().unwrap_or_default(),
96                    block_number,
97                    transaction_hash,
98                })
99            }
100            GolemBaseABI::GolemBaseABIEvents::GolemBaseStorageEntityDeleted(data) => {
101                Ok(Event::EntityRemoved {
102                    entity_id: data.entityKey.into(),
103                    block_number,
104                    transaction_hash,
105                })
106            }
107            GolemBaseABI::GolemBaseABIEvents::GolemBaseStorageEntityBTLExtended(data) => {
108                Ok(Event::EntityExtended {
109                    entity_id: data.entityKey.into(),
110                    old_expiration_block: data.oldExpirationBlock.try_into().unwrap_or_default(),
111                    new_expiration_block: data.newExpirationBlock.try_into().unwrap_or_default(),
112                    block_number,
113                    transaction_hash,
114                })
115            }
116        }
117    }
118}

120/// Client for subscribing to and streaming GolemBase events from the blockchain.
121/// Provides methods to connect to a node and receive event streams for entity changes.
122pub struct EventsClient {
123    provider: DynProvider,
124}
125
126impl EventsClient {
127    /// Creates a new `EventsClient` by connecting to the given websocket `Url`.
128    /// Establishes a connection to the blockchain node for event streaming.
129    pub async fn new(url: Url) -> anyhow::Result<Self> {
130        log::debug!("Connecting to websocket provider: {}", url);
131
132        let provider = ProviderBuilder::new()
133            .connect_ws(WsConnect::new(url.clone()))
134            .await?
135            .erased();
136
137        log::info!("Connected to websocket provider: {}", url);
138        Ok(Self { provider })
139    }
140
141    /// Listens for GolemBase events from the blockchain, starting from the latest block.
142    /// Returns a stream of parsed `Event` items that can be processed asynchronously.
143    pub async fn events_stream<'a>(
144        &'a self,
145    ) -> anyhow::Result<Pin<Box<dyn Stream<Item = anyhow::Result<Event>> + Send + 'a>>> {
146        let filter = self.create_event_filter(BlockNumberOrTag::Latest);
147        self.create_stream_from_filter(filter).await
148    }
149
150    /// Listens for GolemBase events starting from a specific block number.
151    /// Returns a stream of parsed `Event` items from the given block onward.
152    ///
153    /// # Arguments
154    /// * `block` - The block number to start listening for events from.
155    pub async fn events_stream_from_block<'a>(
156        &'a self,
157        block: u64,
158    ) -> anyhow::Result<Pin<Box<dyn Stream<Item = anyhow::Result<Event>> + Send + 'a>>> {
159        let filter = self.create_event_filter(BlockNumberOrTag::Number(block));
160        self.create_stream_from_filter(filter).await
161    }
162
163    /// Creates a filter for GolemBase events, specifying the contract address and event signatures.
164    fn create_event_filter(&self, block: BlockNumberOrTag) -> Filter {
165        Filter::new()
166            .address(GOLEM_BASE_STORAGE_PROCESSOR_ADDRESS)
167            .from_block(block)
168            .events(vec![
169                GolemBaseABI::GolemBaseStorageEntityCreated::SIGNATURE,
170                GolemBaseABI::GolemBaseStorageEntityUpdated::SIGNATURE,
171                GolemBaseABI::GolemBaseStorageEntityDeleted::SIGNATURE,
172                GolemBaseABI::GolemBaseStorageEntityBTLExtended::SIGNATURE,
173            ])
174    }
175
176    /// Creates a stream of events from a filter.
177    /// Subscribes to logs matching the filter and maps them to `Event` values.
178    async fn create_stream_from_filter<'a>(
179        &'a self,
180        filter: Filter,
181    ) -> anyhow::Result<Pin<Box<dyn Stream<Item = anyhow::Result<Event>> + Send + 'a>>> {
182        let subscription = self.provider.subscribe_logs(&filter).await?;
183        Ok(Box::pin(subscription.into_stream().map(Event::try_from)))
184    }
185}
