20/// Removes all existing entities from the GolemBase node
21pub async fn cleanup_entities(client: &GolemBaseClient, account: Address) -> Result<()> {
22    let all_entity_keys = client.get_all_entity_keys().await?;
23    log::info!("Removing all existing entities: {:?}", all_entity_keys);
24    if !all_entity_keys.is_empty() {
25        client.remove_entries(account, all_entity_keys).await?;
26    }
27    Ok(())
28}
