13/// Program to fund and transfer funds between accounts on Golem Base
14#[derive(Parser, Debug)]
15#[command(author, version, about, long_about = None)]
16struct Args {
17    /// URL of the GolemBase node
18    #[arg(short, long, default_value = "http://localhost:8545")]
19    url: String,
20
21    /// Command to execute
22    #[command(subcommand)]
23    command: Command,
24}
25
26#[derive(Subcommand, Debug)]
27enum Command {
28    /// Account management commands
29    Account {
30        #[command(subcommand)]
31        command: AccountCommand,
32    },
33    /// Get entity by ID
34    GetEntity {
35        /// Entity ID to get
36        id: String,
37    },
38    /// Wait until the node is synced
39    WaitSync {
40        /// Timeout in seconds
41        #[arg(short, long, default_value = "120")]
42        timeout: u64,
43    },
44}
45
46#[derive(Debug, Subcommand)]
47pub enum AccountCommand {
48    /// List all available accounts
49    List,
50    /// Create a new account
51    Create {
52        /// Password for the keystore file
53        #[arg(short, long, default_value = "test123")]
54        password: String,
55        /// Save private key in raw format instead of keystore
56        #[arg(long)]
57        raw: bool,
58    },
59    /// Fund an account with ETH
60    Fund {
61        /// The account to fund. If not provided, the default
62        /// private key will be used from raw format.
63        #[arg(long)]
64        wallet: Option<Address>,
65        /// Amount of ETH to send
66        #[arg(long)]
67        amount: BigDecimal,
68    },
69    /// Transfer ETH between accounts
70    Transfer {
71        /// Source account address
72        #[arg(long)]
73        from: Address,
74        /// Destination account address
75        #[arg(long)]
76        to: Address,
77        /// Amount of ETH to send
78        #[arg(long)]
79        amount: BigDecimal,
80        /// Password for the source account
81        #[arg(short, long, default_value = "test123")]
82        password: String,
83    },
84}
85
86impl AccountCommand {
87    async fn execute(&self, client: &GolemBaseClient) -> Result<()> {
88        match self {
89            AccountCommand::List => self.handle_list(client).await,
90            AccountCommand::Create { password, raw } => {
91                self.handle_create(client, password, *raw).await
92            }
93            AccountCommand::Fund { wallet, amount } => {
94                self.handle_fund(client, *wallet, amount.clone()).await
95            }
96            AccountCommand::Transfer {
97                from,
98                to,
99                amount,
100                password,
101            } => {
102                self.handle_transfer(client, *from, *to, amount.clone(), password)
103                    .await
104            }
105        }
106    }
107
108    async fn handle_list(&self, client: &GolemBaseClient) -> Result<()> {
109        let accounts = client.account_sync().await?;
110        println!("Available accounts:");
111        for &addr in &accounts {
112            let balance = client.get_balance(addr).await?;
113            println!("  {}: {} ETH", addr, balance);
114        }
115        Ok(())
116    }
117
118    async fn handle_create(
119        &self,
120        client: &GolemBaseClient,
121        password: &str,
122        raw: bool,
123    ) -> Result<()> {
124        if raw {
125            // Generate a new private key
126            let signer = InMemorySigner::generate();
127            let private_key = signer.private_key().to_bytes();
128            println!(
129                "Generated new private key for address: {}",
130                signer.address()
131            );
132
133            // Get keystore directory and save private key
134            let keystore_dir = InMemorySigner::get_keystore_dir()?;
135            let private_key_path = keystore_dir.join("private.key");
136            println!("Saving private key to: {}", private_key_path.display());
137
138            fs::write(&private_key_path, private_key)?;
139            let address = client.account_register(signer).await?;
140            println!("Account registered with address: {}", address);
141        } else {
142            let account = client.account_generate(password).await?;
143            println!("Created new account: {}", account);
144        }
145        Ok(())
146    }
147
148    async fn handle_fund(
149        &self,
150        client: &GolemBaseClient,
151        wallet: Option<Address>,
152        amount: BigDecimal,
153    ) -> Result<()> {
154        let account = if let Some(wallet) = wallet {
155            // Load account by address
156            client.account_load(wallet, "test123").await?
157        } else {
158            // Load default private key
159            let mut private_key_path =
160                config_dir().ok_or_else(|| anyhow::anyhow!("Failed to get config directory"))?;
161            private_key_path.push("golembase/private.key");
162            client
163                .account_load_file(private_key_path, "test123")
164                .await?
165        };
166        println!("Using account: {account:?}");
167
168        let fund_tx = client.fund(account, amount.clone()).await?;
169        println!("Account funded with {amount} ETH, transaction hash: {fund_tx:?}");
170        Ok(())
171    }
172
173    async fn handle_transfer(
174        &self,
175        client: &GolemBaseClient,
176        from: Address,
177        to: Address,
178        amount: BigDecimal,
179        password: &str,
180    ) -> Result<()> {
181        // Load source account
182        let account = client.account_load(from, password).await?;
183        println!("Using account: {account:?}");
184
185        // Transfer funds
186        let transfer_tx = client.transfer(from, to, amount.clone()).await?;
187        println!(
188            "Transfer transaction hash for {amount} ETH: {:?}",
189            transfer_tx
190        );
191        Ok(())
192    }
193}
194
195impl Command {
196    async fn execute(&self, client: &GolemBaseClient) -> Result<()> {
197        match self {
198            Command::Account { command } => command.execute(client).await,
199            Command::GetEntity { id } => self.handle_get_entity(client, id).await,
200            Command::WaitSync { timeout } => self.handle_wait_sync(client, *timeout).await,
201        }
202    }
203
204    async fn handle_get_entity(&self, client: &GolemBaseClient, id: &str) -> Result<()> {
205        let entry = client.cat(id.parse()?).await?;
206        println!("Entry: {}", entry);
207        Ok(())
208    }
209
210    async fn handle_wait_sync(&self, client: &GolemBaseClient, timeout: u64) -> Result<()> {
211        println!("Waiting for node to sync (timeout: {} seconds)...", timeout);
212        client
213            .sync_node(std::time::Duration::from_secs(timeout))
214            .await?;
215        println!("Node is synced!");
216        Ok(())
217    }
218}

230
231    args.command.execute(&client).await
