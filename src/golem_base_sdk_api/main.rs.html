1use actix_web::{web, App, HttpServer};
2use clap::Parser;
3use dirs::config_dir;
4use golem_base_sdk::{GolemBaseClient, Hash, PrivateKeySigner, Url};
5use sqlx::SqlitePool;
6use tokio::fs;
7
8mod api;
9
10#[derive(Parser, Debug)]
11#[command(author, version, about, long_about = None)]
12struct Cli {
13    #[arg(short, long, default_value = "http://localhost:8545")]
14    rpc_url: String,
15}
16
17#[actix_web::main]
18async fn main() -> Result<(), Box<dyn std::error::Error>> {
19    let cli = Cli::parse();
20    let rpc_url = Url::parse(&cli.rpc_url).expect("Invalid URL format");
21
22    let mut private_key_path = config_dir().ok_or("Failed to get config directory")?;
23    private_key_path.push("golembase/private.key");
24
25    let private_key_bytes = fs::read(&private_key_path).await?;
26    let private_key = Hash::from_slice(&private_key_bytes);
27
28    let signer = PrivateKeySigner::from_bytes(&private_key)
29        .map_err(|e| format!("Failed to parse private key: {}", e))?;
30
31    let client = GolemBaseClient::builder()
32        .wallet(signer)
33        .rpc_url(rpc_url)
34        .build();
35
36    let db_pool = SqlitePool::connect("sqlite::memory:").await?;
37
38    // Ensure the entities table exists
39    sqlx::query(
40        r#"
41        CREATE TABLE IF NOT EXISTS entities (
42            id TEXT PRIMARY KEY,
43            owner TEXT NOT NULL,
44            data TEXT NOT NULL
45        )
46        "#,
47    )
48    .execute(&db_pool)
49    .await?;
50
51    HttpServer::new(move || {
52        App::new()
53            .app_data(web::Data::new(db_pool.clone()))
54            .app_data(web::Data::new(client.clone()))
55            .configure(api::init_routes)
56    })
57    .bind("127.0.0.1:8080")?
58    .run()
59    .await?;
60
61    Ok(())
62}
