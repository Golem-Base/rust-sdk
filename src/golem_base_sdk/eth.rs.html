40/// Represents errors that can occur in the GolemBase ETH client.
41/// Used for wrapping transaction, receipt, and log decoding errors.
42#[derive(Debug, Display, Error)]
43pub enum Error {
44    /// Failed to send transaction: {0}
45    TransactionSendError(String),
46    /// Failed to get transaction receipt: {0}
47    TransactionReceiptError(String),
48    /// Unexpected log data: {0}
49    UnexpectedLogDataError(String),
50}
51
52/// The Ethereum address of the GolemBase storage contract.
53/// All entity-related transactions are sent to this address.
54pub const STORAGE_ADDRESS: Address = address!("0x0000000000000000000000000000000060138453");

56impl GolemBaseClient {
57    pub async fn send_transaction(
58        &self,
59        tx: GolemBaseTransaction,
60    ) -> Result<TransactionResult, Error> {
61        let receipt = self.create_raw_transaction(tx).await?;
62        receipt.try_into()
63    }

65    /// Creates one or more new entities in GolemBase and returns their results.
66    /// Sends a transaction to the storage contract and parses the resulting logs.
67    pub async fn create_entities(&self, creates: Vec<Create>) -> Result<Vec<EntityResult>, Error> {
68        let result = self
69            .send_transaction(GolemBaseTransaction::builder().creates(creates).build())
70            .await;
71
72        result.and_then(|res| match res {
73            TransactionResult {
74                creates,
75                updates,
76                deletes,
77                extensions,
78            } if updates.is_empty() && deletes.is_empty() && extensions.is_empty() => Ok(creates),
79            _ => Err(Error::UnexpectedLogDataError(
80                "Unexpected content in tx logs, this should never happen!".to_string(),
81            )),
82        })
83    }

152    /// NOTE: Nonce management is tricky!
153    /// - This implementation always tries to fetch the latest on-chain nonce before sending a transaction,
154    ///   and only falls back to the locally cached base_nonce if the sync fails.
155    /// - Only the number of in-flight transactions is tracked locally.
156    /// - For robust production use, consider also handling stuck transactions (e.g., gas bumping/EIP-1559).
157    async fn next_nonce(&self) -> u64 {
158        // This is sadly needed because `self.provider.get_transaction_count(self.wallet.address()).pending()`
159        // doesn't give the right number...
160        //
161        //      Error: server returned an error response: error code -32000: replacement transaction underpriced
162        let mut nm = self.nonce_manager.lock().await;
163        let wallet_address = self.wallet.address();
164        match self.provider.get_transaction_count(wallet_address).await {
165            Ok(on_chain_nonce) => {
166                nm.base_nonce = on_chain_nonce;
167            }
168            Err(e) => {
169                log::warn!("Failed to fetch on-chain nonce: {}", e);
170            }
171        }
172        nm.next_nonce().await
173    }

175    /// Creates and sends a raw transaction to the GolemBase storage contract.
176    /// Encodes the transaction payload and sends it to the contract address.
177    pub async fn create_raw_transaction(
178        &self,
179        payload: GolemBaseTransaction,
180    ) -> Result<TransactionReceipt, Error> {
181        log::debug!("payload: {:?}", payload);
182        let encoded = payload.encoded();
183        log::debug!("buffer: {:?}", encoded);
184
185        let nonce = self.next_nonce().await;
186
187        let mut tx = TransactionRequest {
188            to: Some(TxKind::Call(STORAGE_ADDRESS)),
189            input: encoded.into(),
190            chain_id: Some(
191                self.provider
192                    .get_chain_id()
193                    .await
194                    .map_err(|e| Error::TransactionSendError(e.to_string()))?,
195            ),
196            nonce: Some(nonce),
197            ..Default::default()
198        };
199        log::debug!("transaction: {:?}", tx);
200
201        let gas_limit = if let Some(gas_limit) = payload.gas_limit {
202            gas_limit
203        } else {
204            self.provider.estimate_gas(tx.clone()).await.map_err(|e| {
205                Error::TransactionSendError(format!("Failed to estimate gas: {}", e))
206            })?
207        };
208        tx = tx.with_gas_limit(gas_limit);
209
210        if let Some(max_fee_per_gas) = payload.max_fee_per_gas {
211            tx = tx.with_max_fee_per_gas(max_fee_per_gas);
212        }
213
214        if let Some(max_priority_fee_per_gas) = payload.max_priority_fee_per_gas {
215            tx = tx.with_max_priority_fee_per_gas(max_priority_fee_per_gas);
216        }
217
218        let pending_tx = self
219            .provider
220            .send_transaction(tx.clone())
221            .await
222            .map_err(|e| Error::TransactionSendError(e.to_string()))?;
223        log::debug!("pending transaction: {:?}", pending_tx);
224        let receipt = pending_tx
225            .get_receipt()
226            .await
227            .map_err(|e| Error::TransactionReceiptError(e.to_string()))?;
228        log::debug!("receipt: {:?}", receipt);
229        {
230            let mut nm = self.nonce_manager.lock().await;
231            nm.complete().await;
232        }
233
234        if !receipt.status() {
235            self.provider.call(tx).await.map_err(|e| {
236                Error::TransactionReceiptError(format!("Error during tx execution: {e}"))
237            })?;
238        }
239
240        Ok(receipt)
241    }
